func Run()
 sceneState = GetScene("intro" + message)
 fadeState = art.GetFade(fade)
 ?fadeIn
  ?fade < 25 | message = 22 & fade < 35 
   >`@centeredX@,@centeredY@,@fadeState@,@sceneState@
   ?time % 2 = 0
    fade++
  :
   >`@centeredX@,@centeredY@,@fadeState@,@sceneState@
   fadeIn = false
 :
  ?fade > -5
   >`@centeredX@,@centeredY@,@fadeState@,@sceneState@
   ?time % 2 = 0
    fade--
  :
   fadeIn = true
   message ++
   ?message = 23
    global.INTRO = true
    global.phase = "Battle"
