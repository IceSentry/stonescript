func calc_percentage(curVal, maxVal)
  ?maxVal <= 0
    return 0
  var percentage = 0
  percentage = curVal * 100
  percentage = percentage / maxVal
  ?percentage = 0 & curVal > 0
    percentage = 1
  ?percentage < 0
    percentage = 0
  return percentage

func seconds_to_minutes(value)
  ?value < 60
    return value + "s"

  var minutes = value / 60
  var seconds = value % 60
  return minutes + "m" + seconds + "s"
