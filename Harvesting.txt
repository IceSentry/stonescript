func harvest_wood()
  ?res.wood < 1000
    equip hatchet
    ?item.CanActivate()
      activate R
                  
func harvest_stone()
  ?res.stone < 1000
    equip shovel

?harvest.distance < 10
  ?loc = "deadwood"
    harvest_wood()
  :?loc = "rocky"
    harvest_stone()

?pickup.distance < 10
    equipL star
